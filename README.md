import collections

def count_words(text):
    """–ø–æ–¥—Å—á–∏—Ç—ã–≤–∞–µ—Ç –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Å–ª–æ–≤ –≤ —Ç–µ–∫—Å—Ç–µ"""
    return len(text.split())
def most_common_char(text):
    """–Ω–∞—Ö–æ–¥–∏—Ç —Å–∞–º—ã–π —á–∞—Å—Ç–æ –≤—Å—Ç—Ä–µ—á–∞—é—â–∏–π—Å—è —Å–∏–º–≤–æ–ª –≤ —Ç–µ–∫—Å—Ç–µ –∏—Å–∫–ª—é—á–∞—è –ø—Ä–æ–±–µ–ª—ã"""
    counter = collections.Counter(text.replace(' ', ''))
    if counter:
        return counter.most_common(1)[0][0]
    return None  # –µ—Å–ª–∏ —Ç–µ–∫—Å—Ç –ø—É—Å—Ç–æ–π –∏–ª–∏ —Å–æ—Å—Ç–æ–∏—Ç —Ç–æ–ª—å–∫–æ –∏–∑ –ø—Ä–æ–±–µ–ª–æ–≤
def reverse_words(text):
    """–ø–µ—Ä–µ–≤–æ—Ä–∞—á–∏–≤–∞–µ—Ç –∫–∞–∂–¥–æ–µ —Å–ª–æ–≤–æ –≤ —Ç–µ–∫—Å—Ç–µ —Å–æ—Ö—Ä–∞–Ω—è—è –ø–æ—Ä—è–¥–æ–∫"""
    return ' '.join(word[::-1] for word in text.split())
